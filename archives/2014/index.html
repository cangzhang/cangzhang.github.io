获取指定DZ!X主题内容并过滤ubb标签[PHP]

代码如下：

    <?php
    function capword(){
    //连接数据库
    $mysql_server_name='localhost'; 
    $mysql_username='root'; 
    $mysql_password='root';
    $mysql_database='dz';

    $conn=mysql_connect($mysql_server_name,$mysql_username,$mysql_password) or die("error connecting") ; 
    mysql_query("set names 'utf8'"); 
    mysql_select_db($mysql_database); 
    $sql ="SELECT message FROM pre_forum_post WHERE tid=1 and pid=1 and fid=2 and first=1 and invisible=0 ";//pid tid fid确定着具体主题，first代表一楼（即主贴内容）
    $result = mysql_query($sql,$conn) or die('无法连接!'.mysql_error);
    $str;

    //将读取内容赋值$str,并进行过滤
    while( $row = mysql_fetch_array($result) ){
        foreach($row as $value){}
        $str=$value.'';
        //echo $str;
        //echo "<br/>";
        $str=preg_replace('/\[.*?\]/is', '', $str);
        echo substr($str,0,199);//指定输出长度
    }   
    mysql_close($conn);
}
?>
